# TypeScriptлһҖ?

TypeScriptлҠ” JavaScriptм—җ нғҖмһ… мӢңмҠӨн…ңмқ„ м¶”к°Җн•ң м–ём–ҙлӢӨ. JavaScriptмқҳ мғҒмң„ м§‘н•©(Superset)мңјлЎң, TypeScriptлЎң мһ‘м„ұн•ң мҪ”л“ңлҠ” кІ°көӯ JavaScriptлЎң ліҖнҷҳ(м»ҙнҢҢмқј)лҗҳм–ҙ мӢӨн–үлҗңлӢӨ.


# JavaScriptмқҳ н•ңкі„

JavaScriptлҠ” нғҖмһ…мқ„ лӘ…мӢңн•ҳм§Җ м•Ҡм•„лҸ„ лҗҳлҠ” лҸҷм Ғ нғҖмһ… м–ём–ҙлӢӨ. мқҙ мң м—°н•Ёмқҙ мҳӨнһҲл Ө л¬ём ңк°Җ лҗҳлҠ” кІҪмҡ°к°Җ л§ҺлӢӨ.

// JavaScript вҖ” нғҖмһ… мҳӨлҘҳлҘј мӢӨн–ү м „м—җ м•Ң мҲҳ м—ҶмқҢ
function add(a, b) {
  return a + b;
}

console.log(add(5, "3")); // 53 вҶ’ мқҳлҸ„мҷҖ лӢӨлҘё кІ°кіјм§Җл§Ң мҳӨлҘҳк°Җ лӮҳм§Җ м•ҠмқҢ

н”„лЎңм қнҠёк°Җ м»Өм§ҲмҲҳлЎқ мқҙлҹ° мӢӨмҲҳк°Җ мҢ“мқҙкі , м–ҙл””м„ң мһҳлӘ»лҗң к°’мқҙ л“Өм–ҙмҷ”лҠ”м§Җ м¶”м Ғн•ҳкё° м–ҙл ӨмӣҢм§„лӢӨ.


# TypeScriptлҘј м“°лҠ” мқҙмң 

// TypeScript вҖ” мһҳлӘ»лҗң нғҖмһ…мқ„ мҪ”л“ң мһ‘м„ұ мӨ‘м—җ л°”лЎң к°җм§Җ
function add(a: number, b: number): number {
  return a + b;
}

add(5, "3"); // м»ҙнҢҢмқј м—җлҹ¬ вҶ’ мӢӨн–ү м „м—җ л°”лЎң м•Ң мҲҳ мһҲмқҢ
  • мӢӨн–ү м „м—җ мҳӨлҘҳлҘј мһЎмқ„ мҲҳ мһҲлӢӨ вҶ’ л°°нҸ¬ нӣ„ лІ„к·ёлҘј мӨ„мқј мҲҳ мһҲмқҢ
  • мҪ”л“ң мһҗмІҙк°Җ л¬ём„ңк°Җ лҗңлӢӨ вҶ’ н•ЁмҲҳлҘј ліҙл©ҙ м–ҙл–Ө к°’мқ„ л°ӣкі  л°ҳнҷҳн•ҳлҠ”м§Җ л°”лЎң м•Ң мҲҳ мһҲмқҢ
  • IDE мһҗлҸҷмҷ„м„ұмқҙ к°•л Ҙн•ҙм§„лӢӨ вҶ’ нғҖмһ… м •ліҙлҘј кё°л°ҳмңјлЎң м •нҷ•н•ң мһҗлҸҷмҷ„м„ұ м ңкіө
  • нҢҖ нҳ‘м—…м—җ мң лҰ¬н•ҳлӢӨ вҶ’ лӢӨлҘё мӮ¬лһҢмқҙ м§  мҪ”л“ңлҸ„ нғҖмһ…л§Ң ліҙл©ҙ мӮ¬мҡ©лІ•мқ„ м•Ң мҲҳ мһҲмқҢ

# TypeScript мӢӨн–ү кіјм •

TypeScriptлҠ” лёҢлқјмҡ°м ҖлӮҳ Node.jsк°Җ м§Ғм ‘ мӢӨн–үн•  мҲҳ м—Ҷкё° л•Ңл¬ём—җ tsc(TypeScript Compiler)лЎң JavaScriptлЎң ліҖнҷҳн•ҳлҠ” кіјм •мқҙ н•„мҡ”н•ҳлӢӨ.

TypeScript мһ‘м„ұ (.ts)
    вҶ“
tscлЎң м»ҙнҢҢмқј (tsc main.ts)
    вҶ“
JavaScript мғқм„ұ (.js)
    вҶ“
Node.js лҳҗлҠ” лёҢлқјмҡ°м Җм—җм„ң мӢӨн–ү (node main.js)

# м–ём ң TypeScriptлҘј м“°лҠ”к°Җ

  • к·ңлӘЁк°Җ нҒ° н”„лЎңм қнҠё (нҢҢмқј, н•ЁмҲҳк°Җ л§Һм•„м§ҲмҲҳлЎқ нғҖмһ…мқҳ к°Җм№ҳк°Җ лҶ’м•„м§җ)
  • нҢҖ нҳ‘м—… (м—¬лҹ¬ лӘ…мқҙ к°ҷмқҖ мҪ”л“ңлҘј мҲҳм •н•  л•Ң нғҖмһ…мқҙ м•Ҳм „л§қ м—ӯн• )
  • мһҘкё°м ҒмңјлЎң мң м§ҖліҙмҲҳн•ҙм•ј н•ҳлҠ” м„ңл№„мҠӨ
  • React, Next.js л“ұ нҳ„лҢҖ н”„лЎ нҠём—”л“ң н”„л Ҳмһ„мӣҢнҒ¬мҷҖ н•Ёк»ҳ мӮ¬мҡ©н•  л•Ң


# вң… 1. мӣҗмӢң нғҖмһ… (Primitive Types)

TypeScriptмқҳ мӣҗмӢң нғҖмһ…мқҖ лӢӨмқҢкіј к°ҷлӢӨ.

нғҖмһ… м„ӨлӘ… мҳҲмӢң
string л¬ёмһҗм—ҙ "hello"
number мҲ«мһҗ (м •мҲҳ, мӢӨмҲҳ лӘЁл‘җ) 42, 3.14
boolean м°ё/кұ°м§“ true, false
null мқҳлҸ„м ҒмңјлЎң к°’мқҙ м—ҶмқҢмқ„ лӘ…мӢң null
undefined м•„м§Ғ к°’мқҙ н• лӢ№лҗҳм§Җ м•ҠмқҢ undefined
symbol кі мң н•ң мӢқлі„мһҗ Symbol("id")

мӣҗмӢң нғҖмһ…мқ„ м ңмҷён•ң лӘЁл“  к°’мқҖ objectлЎң л¶„лҘҳлҗңлӢӨ.


# нғҖмһ… м„ м–ё л°©лІ•

// м„ м–ёкіј лҸҷмӢңм—җ мҙҲкё°к°’ м—Ҷмқҙ нғҖмһ…л§Ң м§Җм •
let userName: string;
let userAge: number;
let isAdmin: boolean;

// лӮҳмӨ‘м—җ к°’ н• лӢ№
userName = "Alice";
userAge = 25;
isAdmin = true;

// м„ м–ёкіј лҸҷмӢңм—җ мҙҲкё°к°’ м§Җм •
let productName: string = "л§Ҙл¶Ғ н”„лЎң";
let productPrice: number = 3000000;
let isAvailable: boolean = true;

# н•ЁмҲҳмқҳ л§Өк°ңліҖмҲҳмҷҖ л°ҳнҷҳ нғҖмһ… м§Җм •

// л§Өк°ңліҖмҲҳмҷҖ л°ҳнҷҳк°’м—җ нғҖмһ… м§Җм •
const addNumbers = (a: number, b: number): number => {
  return a + b;
};

console.log(addNumbers(5, 3)); // 8


# вң… 2. nullкіј undefinedмқҳ м°Ёмқҙ

л‘ҳ лӢӨ "к°’мқҙ м—ҶлҠ” мғҒнғң"лҘј лӮҳнғҖлӮҙм§Җл§Ң мқҳлҜёк°Җ лӢӨлҘҙлӢӨ.

let a: undefined; // м•„м§Ғ к°’мқҙ н• лӢ№лҗҳм§Җ м•ҠмқҖ мғҒнғң
let b: null = null; // мқҳлҸ„м ҒмңјлЎң к°’мқҙ м—ҶмқҢмқ„ лӘ…мӢңн•ң мғҒнғң


# вң… 3. typeofлЎң нғҖмһ… нҢҗлі„ мӢң мЈјмқҳм җ

JavaScript(к·ёлҰ¬кі  TypeScript)м—җм„ң typeof nullмқҖ "object"лҘј л°ҳнҷҳн•ңлӢӨ. мқҙлҠ” JavaScript мҙҲкё° м„Өкі„ лӢ№мӢңмқҳ кө¬нҳ„ мҳӨлҘҳлЎң, н•ҳмң„ нҳёнҷҳм„ұ л•Ңл¬ём—җ м§ҖкёҲк№Ңм§Җ мҲҳм •лҗҳм§Җ м•Ҡкі  мһҲлӢӨ.

typeof "hello"; // "string"
typeof 42; // "number"
typeof true; // "boolean"
typeof undefined; // "undefined"
typeof null; // "object" вҶҗ мӣҗмӢң нғҖмһ…мқҙм§Җл§Ң objectк°Җ л°ҳнҷҳлҗЁ (мЈјмқҳ!)
typeof {}; // "object"
typeof []; // "object"

л”°лқјм„ң typeofлЎң мӣҗмӢң нғҖмһ…мқ„ нҢҗлі„н•  л•ҢлҠ” nullмқ„ л°ҳл“ңмӢң мҳҲмҷё мІҳлҰ¬н•ҙм•ј н•ңлӢӨ.

// nullмқ„ мҳҲмҷё мІҳлҰ¬н•ҙм„ң мӣҗмӢң нғҖмһ… м—¬л¶ҖлҘј м •нҷ•нһҲ нҢҗлі„
const isPrimitive = (value: unknown): boolean => {
  if (value === null || typeof value !== "object") {
    return true;
  }
  return false;
};

console.log(isPrimitive("Hello")); // true
console.log(isPrimitive(42)); // true
console.log(isPrimitive(null)); // true  вҶҗ null мҳҲмҷё мІҳлҰ¬ лҚ•л¶„м—җ м •нҷ•нһҲ нҢҗлі„
console.log(isPrimitive({})); // false
console.log(isPrimitive([])); // false


# вң… 4. мң лӢҲмҳЁ нғҖмһ…мңјлЎң м—¬лҹ¬ нғҖмһ… н—Ҳмҡ©

|лҘј мӮ¬мҡ©н•ҳл©ҙ м—¬лҹ¬ нғҖмһ… мӨ‘ н•ҳлӮҳлҘј н—Ҳмҡ©н•  мҲҳ мһҲлӢӨ.

// string, null, undefined мӨ‘ н•ҳлӮҳлҘј л°ӣмқ„ мҲҳ мһҲлҠ” н•ЁмҲҳ
const stringifyValue = (value: string | null | undefined): string => {
  if (typeof value === "string") {
    return value;
  }
  return "к°’мқҙ м—ҶмҠөлӢҲлӢӨ";
};

console.log(stringifyValue("Hello")); // "Hello"
console.log(stringifyValue(null)); // "к°’мқҙ м—ҶмҠөлӢҲлӢӨ"
console.log(stringifyValue(undefined)); // "к°’мқҙ м—ҶмҠөлӢҲлӢӨ"


# вң… 5. == vs === (лҠҗмҠЁн•ң лҸҷл“ұм„ұ vs м—„кІ©н•ң лҸҷл“ұм„ұ)

===лҠ” нғҖмһ…кіј к°’мқҙ лӘЁл‘җ к°ҷм•„м•ј trueлҘј л°ҳнҷҳн•ңлӢӨ. нғҖмһ…мқҙ лӢӨлҘҙл©ҙ к°’мқҙ к°ҷм•„лҸ„ falseлӢӨ.

5 === 5; // true
5 === "5"; // false вҶ’ мҲ«мһҗ 5мҷҖ л¬ёмһҗм—ҙ "5"лҠ” нғҖмһ…мқҙ лӢӨлҘҙкё° л•Ңл¬ё

==лҠ” нғҖмһ…мқҙ лӢ¬лқјлҸ„ лӮҙл¶Җм ҒмңјлЎң нғҖмһ… ліҖнҷҳмқ„ мӢңлҸ„н•ң нӣ„ к°’мқ„ л№„көҗн•ңлӢӨ. к·ёлһҳм„ң мқҳлҸ„н•ҳм§Җ м•ҠмқҖ кІ°кіјк°Җ лӮҳмҳ¬ мҲҳ мһҲлӢӨ.

5 == "5"; // true вҶ’ л¬ёмһҗм—ҙ "5"к°Җ мҲ«мһҗ 5лЎң ліҖнҷҳлҗң нӣ„ л№„көҗлҗЁ

# мЈјмқҳн•ҙм•ј н•  нҠ№мҲҳ мјҖмқҙмҠӨ

nullкіј undefinedлҠ” л‘ҳ лӢӨ "к°’мқҙ м—ҶмқҢ"мқ„ мқҳлҜён•ҳлҠ” мғҒнғңмқҙкё° л•Ңл¬ём—җ == л№„көҗм—җм„ңлҠ” к°ҷлӢӨкі  м·ЁкёүлҗңлӢӨ. н•ҳм§Җл§Ң нғҖмһ… мһҗмІҙк°Җ лӢӨлҘҙкё° л•Ңл¬ём—җ ===м—җм„ңлҠ” falseлӢӨ.

null == undefined; // true
null === undefined; // false

NaNмқҖ "мҲ«мһҗк°Җ м•„лӢҳ(Not a Number)"мқ„ мқҳлҜён•ҳлҠ” нҠ№мҲҳн•ң к°’мңјлЎң, мһҗкё° мһҗмӢ мқ„ нҸ¬н•Ён•ң м–ҙл–Ө к°’кіј л№„көҗн•ҙлҸ„ н•ӯмғҒ falseлҘј л°ҳнҷҳн•ңлӢӨ.

NaN == NaN; // false
NaN === NaN; // false

мӢӨл¬ҙм—җм„ңлҠ” мҳҲмғҒм№ҳ лӘ»н•ң нғҖмһ… ліҖнҷҳмңјлЎң мқён•ң лІ„к·ёлҘј л°©м§Җн•ҳкё° мң„н•ҙ ===лҘј мӮ¬мҡ©н•ҳлҠ” кІғмқҙ мӣҗм№ҷмқҙлӢӨ.


н•өмӢ¬ мҡ”м•Ҫ:

  • TypeScript вҶ’ JavaScriptм—җ нғҖмһ…мқ„ м¶”к°Җн•ң м–ём–ҙ, мӢӨн–ү м „м—җ мҳӨлҘҳлҘј мһЎмқ„ мҲҳ мһҲмқҢ
  • мӣҗмӢң нғҖмһ… вҶ’ string, number, boolean, null, undefined, symbol
  • typeof null вҶ’ "object" л°ҳнҷҳ (м„Өкі„ мҳӨлҘҳ), nullмқҖ л°ҳл“ңмӢң мҳҲмҷё мІҳлҰ¬ н•„мҡ”
  • мң лӢҲмҳЁ нғҖмһ… вҶ’ |лЎң м—¬лҹ¬ нғҖмһ…мқ„ н—Ҳмҡ©
  • === вҶ’ нғҖмһ…кіј к°’ лӘЁл‘җ л№„көҗ, мӢӨл¬ҙм—җм„ңлҠ” === мӮ¬мҡ©мқҙ мӣҗм№ҷ

вҶҗ / TypeScript нғҖмһ…кіј мқён„°нҺҳмқҙмҠӨ м •лҰ¬ вҶ’
